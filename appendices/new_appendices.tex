\section{Appendix Z1: $d=5$ uniqueness theorem}
\label{app:d5}

\begin{theorem}[$d=5$ uniqueness, Tier~1]\label{thm:d5unique}
Among all dimensions $d\geq 1$, the following four conditions are
satisfied simultaneously if and only if $d=5$:
\begin{enumerate}[label=\upshape(C\arabic*)]
\item\label{C1} The Hosotani prefactor $P_d=2\,d_S\,\Gamma(d/2)\,\pi^{2-d/2}$
  is a positive integer. \emph{Restricts to $d\in\{4,5\}$.}
\item\label{C2} $\pi_d(\mathrm{SU}(3))\cong\mathbb{Z}$ (infinite cyclic).
  \emph{Restricts to $d\in\{3,5\}$.}
\item\label{C3} $N_c^2-1=2^{\lceil d/2\rceil}$ admits an integer solution
  $N_c\geq 2$. \emph{Restricts to $d\in\{5,6\}$ with $N_c=3$.}
\item\label{C4} $d$ is odd (no perturbative gauge anomalies).
\end{enumerate}
Intersection: $\{4,5\}\cap\{3,5\}\cap\{5,6\}\cap\{\text{odd}\}=\{5\}$.
Moreover, $d=5$ uniquely determines $N_c=3$ and $P_5=6=|W(\mathrm{SU}(3))|$.
\end{theorem}

\begin{proof}
\ref{C1}: For odd $d$, $P_d$ contains $\pi^{(5-d)/2}$, which is rational
only if $d=5$. For even~$d$, $P_d$ contains $\pi^{2-d/2}$, rational only
if $d=4$. \ref{C2}: Standard homotopy tables; $\pi_{2k+1}(\mathrm{SU}(N))=\mathbb{Z}$
for $k\leq N-1$, giving $d\in\{3,5\}$ for $N=3$.
\ref{C3}: $2^n+1=m^2$ has the unique solution $n=3$, $m=3$ (since
$k(k+1)=2^{n-2}$ with coprime $k,k+1$ forces $k=1$).
Thus $\lceil d/2\rceil=3$, i.e.\ $d\in\{5,6\}$.
\ref{C4}: trivially restricts to $d$~odd.
Verified computationally for $d=1,\ldots,30$ (\texttt{d5\_uniqueness.py}).
\end{proof}

\label{app:ramond}

\begin{theorem}[Ramond BC uniqueness, Tier~1]\label{thm:ramond}
On $T^5_R$ with SU(3) gauge field, fermion boundary conditions are
uniquely Ramond (periodic) in all~5 directions.

\begin{proof}
Three independent arguments:

\emph{(R1) Permutation symmetry.}
Conditions~\ref{C1}--\ref{C4} of Theorem~\ref{thm:d5unique} treat all~5
directions identically. No mechanism distinguishes a ``Ramond direction''
from an ``NS direction.'' Consistency requires all BCs equal: either all
Ramond or all~NS.

\emph{(R2) Sign of the instanton correction.}
Experimentally $m_p/m_e>6\pi^5$, requiring a positive correction.
At the self-dual point, the Poisson-dual mode $m=1$ contributes
$+2\varepsilon$ (Ramond) or $-2\varepsilon$ (NS) per dimension, with
$\varepsilon=e^{-\pi^2}$. Only Ramond gives a positive 5D correction.

\emph{(R3) Hosotani saturation.}
With Ramond~BC the Hosotani potential has its minimum at $a=1/2$
(Appendix~\ref{app:hosotani}), producing the saturation mechanism.
NS~BC gives a trivial minimum at $a=0$.

Each pair (R1)+(R2) and (R1)+(R3) independently selects pure Ramond.
Verified numerically (\texttt{ramond\_bc\_proof.py}).
\end{proof}

\begin{theorem}[Quark charges, Tier~1]\label{thm:quarks}
Given SU(3) with quarks in the fundamental representation, and the
experimental facts $Q_p=+1$ (proton) and $Q_n=0$ (neutron), the
quark charges $Q_u=2/3$ and $Q_d=-1/3$ are uniquely determined.
\end{theorem}

\begin{proof}
The proton $uud$ and neutron $udd$ give $2Q_u+Q_d=1$ and $Q_u+2Q_d=0$.
Unique solution: $Q_u=2/3$, $Q_d=-1/3$.
\end{proof}

\begin{corollary}[$\alpha^1$-cancellation as corollary]\label{cor:alpha1}
$\sum_{\mathrm{proton}}Q_i^2 = 2(2/3)^2+(-1/3)^2=1=Q_e^2$ follows
automatically. It is not an independent constraint.
\end{corollary}

\begin{corollary}[$N_c=3$ from charge cancellation]\label{cor:Nc3charge}
For general SU($N_c$) with baryons of~$N_c$ quarks, the condition
$\sum_p Q_i^2=Q_e^2=1$ (given $Q_p=1$, $Q_n=0$) is satisfied \emph{only}
for $N_c=3$. This is a fourth independent proof of $N_c=3$.
\end{corollary}

\section{Appendix Z9: Structural motivation for Axiom~M}
\label{app:axiomM_structure}

Axiom~M cannot be derived from first principles (it connects UV spectral
geometry to IR hadron masses across $\sim\!20$ orders of magnitude).
However, its \emph{structure} is uniquely determined by five general
principles.

\begin{theorem}[Structural uniqueness of Axiom~M, Tier~1]
\label{thm:axiomM_struct}
Any mass-ratio map from the spectral geometry of $T^5$ with $\mathrm{SU}(3)$
to 4D hadron masses that satisfies:
\begin{enumerate}[label=\upshape(P\arabic*),nosep]
\item \emph{Spectral origin:} masses arise from $\zeta$-function determinants
  of the Dirac operator;
\item \emph{Multiplicativity:} independent toroidal directions contribute via
  heat-kernel factorization $K(t)=\prod_\mu K_\mu(t)$;
\item \emph{$R$-independence:} the ratio $m_p/m_e$ does not depend on the
  compactification radius (Appendix~\ref{app:Rindep});
\item \emph{Topological quantization:} winding numbers from
  $\pi_5(\mathrm{SU}(3))=\mathbb{Z}$ are integers;
\item \emph{Gauge covariance:} the result is $\mathrm{SU}(3)$-invariant,
  contributing the Weyl-group factor $|W|$;
\end{enumerate}
must have the form
\begin{equation}\label{eq:axiomM_structure}
  \frac{m_p}{m_e}
  = |W|\cdot\bigl[\bar{K}(\sigma^*)\bigr]^{n_p-n_e},
  \qquad n_p-n_e\in\mathbb{Z},
\end{equation}
up to a normalization convention for~$\bar{K}$.
\end{theorem}

\begin{proof}[Proof sketch]
(P1) gives $\ln(m_p/m_e)$ linear in spectral data.
(P2) identifies the data as the 5D heat kernel
$K(\sigma^*)=\Theta_3(1)^5$.
(P3) restricts the argument to $\sigma^*/R^2=1$.
(P4) forces the exponent to be an integer.
(P5) introduces the overall factor $|W|=6$.
Combining yields~\eqref{eq:axiomM_structure}.
See \texttt{axiom\_m\_structure.py} for numerical verification.
\end{proof}

The \emph{residual content} of Axiom~M --- what P1--P5 do \emph{not}
determine --- is:
\begin{itemize}[nosep]
\item the normalization of $\bar{K}$ (which sets the leading term $6\pi^5$);
\item the specific winding numbers $n_p=2$, $n_e=0$;
\item the instanton corrections ($\alpha^2/\!\sqrt{8}$).
\end{itemize}
The integer $n_p-n_e=2$ is the \emph{unique} value giving the correct
order of magnitude: $\Delta n=1$ gives $105$ (too small by $17\times$),
$\Delta n=3$ gives $32{,}120$ (too large by $17\times$).

\paragraph{Analogy.}
In general relativity, Lovelock's theorem determines the \emph{structure}
of the field equation (second-order, symmetric, divergence-free) from
general principles, while the cosmological constant~$\Lambda$ remains
a free parameter. Axiom~M is analogous: its form follows from P1--P5,
its numerical content (which yields $6\pi^5$) does not.
