\section{Zeta-residue bookkeeping}
%% ================================================================
For a Laplace- or Dirac-type spectral problem on a compact $d$-dimensional manifold,
the heat kernel has the formal asymptotic expansion
$K(t)\sim\sum_{k\geq0}a_k\,t^{(k-d)/2}$,
and $a_k t^{(k-d)/2}$ generates a pole at $s=(d-k)/2$.

For $d=5$, the pole at $s=5/2$ is governed by $a_0$ alone.  On the flat torus,
$a_0\propto\operatorname{rank}(V)\sqrt{\det g}$, so under volume-preserving
deformations $R_j=\bar R(1+\epsilon_j)$, $\sum_j\epsilon_j=0$:
\[
  \prod_{j=1}^5(1+\epsilon_j) = 1+\mathcal{O}(\epsilon^2).
\]
The leading residue is unchanged to first order.

quotient.

\section{The $\pi^7$-bridge --- six approaches and their failure}
\label{sec:pi7bridge}
%% ================================================================

The ``$\pi^7$-bridge'' refers to the connection
$6/\pi^2 \xrightarrow{\;?\;} 6\pi^5$.
This requires a mechanism $m\propto 1/V$ (soliton mass proportional to
inverse spectral volume), giving $\pi^{-2}\times\pi^7=\pi^5$.
Six independent spectral-geometric approaches were tested; all fail.

\begin{enumerate}[leftmargin=1.5em]
\item \textbf{Hosotani barrier height.}  $\Delta V_H = V_H(0)-V_H(1/2)$
  is exponentially small ($\propto e^{-\pi^2}\approx 5\times10^{-5}$),
  as is typical for thermal-type corrections on the torus.  The baseline
  value $6\pi^5\approx 1836$ is algebraic, not exponential; these live on
  different scales.  \textbf{Fails.}

\item \textbf{Eigenvalue quotient.}  The ratio of the second to the first
  nonzero eigenvalue of $\Delta_0$ on $T^5_{R=1/2}$ is~4 (for the $|n|^2=1$ to $|n|^2=0$
  modes), not a power of~$\pi$.

\item \textbf{$\zeta$-determinant ratio.}  $\det'(\Delta_p)/\det'(\Delta_e)
  \approx e^{-1.08}$, far from~$6\pi^5$.

\item \textbf{Orbifold $T^5/\mathbb{Z}_3$.}  The twisted sector contribution
  to the one-loop determinant is $\Delta Z_4'=-16/243$ (rational, exact),
  not related to $\pi^7$.

\item \textbf{'t~Hooft twist.}  A twisted boundary condition producing a
  $\mathbb{Z}_3$ magnetic flux contributes a factor~3 to the partition function,
  not $\pi^7$.

\item \textbf{Saddle-point approximation.}  A saddle-point estimate of the
  instanton density on $T^5$ gives $p\approx6.15$ (close to~6, but the
  $\pi^5$ part is absent; the prefactor is $\mathcal{O}(1)$).
\end{enumerate}

\textbf{Structural conclusion.}  Free spectral geometry on $T^5$ cannot
generate hierarchies of $\mathcal{O}(10^3)$.  The factor $\pi^7$ most likely
requires non-perturbative QCD dynamics (confinement on the torus), consistent
with the fact that 99\% of the proton mass arises from QCD binding energy, not
from geometric or perturbative-EM effects.

\section{The 2-instanton mechanism for $\alpha^2$}
\label{sec:2instanton}

The quadratic nature of the electromagnetic correction admits a direct
topological interpretation via a 2-instanton configuration.

\paragraph{Geometric setup.}
Write $T^5=T^2_{12}\times T^2_{34}\times S^1_5$, giving two independent

\paragraph{Single-instanton contribution.}

$\alpha_0$ is identified with the physical fine-structure constant~$\alpha$

\section{Appendix Y: Derivation of $c_2$ via Elizalde $\zeta$-regularization}
\label{sec:elizalde}
%% ================================================================

The 2-loop coefficient $c_2$ in the $\alpha$-relation is determined by
the 1-loop functional determinant of the covariant Laplacian in the
instanton background on $T^2\times T^3$.

\subsection{Spectrum}

On $T^2$ with one unit of magnetic flux, the eigenvalues are Landau
levels $\lambda_n=2B(n+\tfrac12)$, $n=0,1,2,\ldots$\, with degeneracy
$d_B=1$.  On $T^3$ with Ramond boundary conditions, the eigenvalues are
$\beta_{\mathbf{k}}=\mu^2\sum_{i=1}^3(k_i+\tfrac12)^2$,
$k_i\in\mathbb{Z}$.  Both carry Hurwitz parameter $a=\tfrac12$.

\subsection{Heat kernel factorization}

The combined operator $\Lambda=A\otimes\mathbb{1}+\mathbb{1}\otimes B$
gives $K(t)=K_{T^2}(t)\cdot K_{T^3}(t)$, and the spectral
$\zeta$-function via Mellin transform:
\[
  \zeta(s) = \frac{1}{\Gamma(s)}\int_0^\infty t^{s-1}\,
  K_{T^2}(t)\cdot K_{T^3}(t)\,dt\,.
\]

\subsection{The key identity (Tier~1)}

The Hurwitz $\zeta$-function at $a=\tfrac12$ satisfies
$\zeta_H(s,\tfrac12) = (2^s-1)\zeta_R(s)$, from which
$\zeta_H(0,\tfrac12)=0$ and
\begin{equation}\label{eq:hurwitz}
  \zeta_H'(0,\tfrac12) = \ln 2\cdot\zeta_R(0) = -\tfrac12\ln 2\,,
\end{equation}
since $\zeta_R(0)=-\tfrac12$.  This is exact and universally valid.

\subsection{Per-dimension counting (Tier~1)}

Each of the $d=5$ compact dimensions (two from $T^2$ Landau levels,
three from $T^3$ Ramond modes) carries Hurwitz parameter $a=\tfrac12$.
The Elizalde product formula for $\zeta'(0)$ on a product manifold
assigns to each dimension a weight $\zeta_H'(0,\tfrac12)=-\tfrac12\ln 2$.
Total: coefficient of $\ln 2 = d/2 = 5/2$.

\subsection{Epstein cross-term (Tier~1)}

The Elizalde formula $\zeta'_{A+B}(0)=\zeta_A(0)\zeta_B'(0)+
\zeta_A'(0)\zeta_B(0)+\mathcal{R}$ has vanishing leading terms (because
$\zeta_H(0,\tfrac12)=0$), so the cross-term $\mathcal{R}$ dominates.
From the subleading heat-kernel coefficients on $T^3$, the
proton--electron difference evaluates to
$\mathcal{R}_{\mathrm{p}}-\mathcal{R}_{\mathrm{e}}
=-(d{-}2)/\dim_\mathbb{R}(S)=-3/8$, where $d{-}2=3$ counts the
$T^3$ dimensions and $\dim_\mathbb{R}(S)=8$.

\subsection{Result}

\[
  c_2 = \frac{d}{2}\ln 2 - \frac{d-2}{\dim_\mathbb{R}(S)}
  = \frac{5}{2}\ln 2 - \frac{3}{8} \approx 1.35787\,.
\]
